Version 1 
Script stored in tree part and spawns bug and moves it to different tree

Code: 
local listOfBugs = {}
local TweenService = game:GetService("TweenService")
local tree = script.Parent
local allTrees = {workspace["Tree 1"], workspace["Tree 2"], workspace["Tree 3"], workspace["Tree 4"], workspace["Tree 5"], workspace["Tree 6"], workspace["Tree 7"], workspace["Tree 8"], workspace["Tree 9"], workspace["Tree 10"]}

for i = 1,10 do
	local bug = Instance.new("Part")
	bug.Name = "Bug " .. i
	bug.Anchored = true
	bug.Shape = Enum.PartType.Ball
	bug.Size = Vector3.new(3, 3, 3)
	bug.Parent = workspace
	bug.Position = tree.Position + Vector3.new(5, 10, 0)
	table.insert(listOfBugs, bug)
end

for i in pairs(listOfBugs) do 
	
	local tween = TweenService:Create(
		listOfBugs[i],
		TweenInfo.new(2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
		{ Position = allTrees[math.random(1, #allTrees)].Position + Vector3.new(5, 10, 0) }
	)
	tween:Play()
	tween.Completed:Wait()
	listOfBugs[i]:Destroy()
	
end
